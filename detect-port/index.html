如何自动获取可用端口号

灵魂拷问

create-reate-app 生成的项目，在启动时，发现端口号被占用，会自动切换到一个可用的新端口号，这是如何实现的？

detect-port

detect-port 库，用 JavaScript 实现的端口检测器。

在 create-react-app 中，启动项目时使用的 react-scripts 库中，就用到了这个库，当默认 3000 端口号被占用时，会询问是否使用 3001 做端口号？这是怎么做到的呢？

detect-port 的使用

使用他有两种方式：

1. 命令行

全局安装

$ npm i detect-port -g

通过命令行使用

# 获取一个随机的可用的端口号
$ detect

# 检测指定端口号
$ detect 80

# 在命令后增加 --verbose 参数，可以获取详情的日志信息
$ detect --verbose // get available port 23577 randomly

# 更多帮助
$ detect --help

2.函数调用

本地安装

$ npm i detect-port --save
  1. 使用回调函数
const detect = require('detect-port');

detect(port, (err, _port) => { // _port 是得到的可用的端口号，可能与 port 不一致
  if (err) {
    console.log(err);
  }

  if (port == _port) {
    console.log(`port: ${port} was not occupied`);
  } else {
    console.log(`port: ${port} was occupied, try port: ${_port}`);
  }
});
  1. 使用 yield 语法
const co = require('co');

co(function *() {
  const _port = yield detect(port); // _port 是得到的可用的端口号，可能与 port 不一致

  if (port == _port) {
    console.log(`port: ${port} was not occupied`);
  } else {
    console.log(`port: ${port} was occupied, try port: ${_port}`);
  }
});
  1. 使用 promise
detect(port)
  .then(_port => { // _port 是得到的可用的端口号，可能与 port 不一致
    if (port == _port) {
      console.log(`port: ${port} was not occupied`);
    } else {
      console.log(`port: ${port} was occupied, try port: ${_port}`);
    }
  })
  .catch(err => {
    console.log(err);
  });

实现原理

来看看里面的源码吧，看看是怎么判断当前端口是否可用的？当前端口号不可用，又怎么返回一个新的可用的端口号？

第一步，参数处理

port，host，callback 都是可选参数。

module.exports = (port, host, callback) => {
  if (typeof port === 'function') {
    callback = port;
    port = null;
  } else if (typeof host === 'function') {
    callback = host;
    host = null;
  }
  port = parseInt(port) || 0; // 端口号默认是 0
  let maxPort = port + 10;
  if (maxPort > 65535) { // 对端口号做最大限制
    maxPort = 65535;
  }
  debug('detect free port between [%s, %s)', port, maxPort);
  if (typeof callback === 'function') { // callback 如果是函数，就用回调函数的方式返回。
    return tryListen(host, port, maxPort, callback);
  }
  // promise
  return new Promise((resolve, reject) => { // callback 如果不是函数，就用 promise 的方式返回。
    tryListen(host, port, maxPort, (error, realPort) => {
      if (error) {
        reject(error);
      } else {
        resolve(realPort);
      }
    });
  });
};

做检查端口的逻辑，在 tryListen 中：

在 tryListen 中，定义了 handleError 方法，如果当前端口号不可用，就会进入此函数，将端口号 +1，再继续使用 tryListen 检测新端口号是否可用。

循环往复~

function tryListen(host, port, maxPort, callback) {
  function handleError() {
    port++;
    if (port >= maxPort) {
      debug(
        'port: %s >= maxPort: %s, give up and use random port',
        port,
        maxPort
      );
      port = 0;
      maxPort = 0;
    }
    tryListen(host, port, maxPort, callback);
  }

  // 1. check specified host (or null)
  listen(port, host, (err, realPort) => {
    // ignore random listening
    if (port === 0) {
      return callback(err, realPort);
    }

    if (err) {
      return handleError(err);
    }

    // 2. check default host
    listen(port, null, err => {
      if (err) {
        return handleError(err);
      }

      // 3. check localhost
      listen(port, 'localhost', err => {
        if (err) {
          return handleError(err);
        }

        // 4. check current ip
        let ip;
        try {
          ip = address.ip();
        } catch (err) {
          // Skip the `ip` check if `address.ip()` fails
          return callback(null, realPort);
        }

        listen(port, ip, (err, realPort) => {
          if (err) {
            return handleError(err);
          }

          callback(null, realPort);
        });
      });
    });
  });
}

listen 函数，是用来判断当前端口是否可用的。

在上面的 tryListen 中，嵌套了四层 listen，分别对同一 port，不同 host 的场景，做检测：

  1. port, host
  2. port, null
  3. port, 'localhost'
  4. port, ip

只有当上面四种场景，都检测通过后，才算成功。

若有一种情况失败，就会进入 handleError 中，port + 1，再次循环检测。

function listen(port, hostname, callback) {
  const server = new net.Server();

  server.on('error', err => {
    debug('listen %s:%s error: %s', hostname, port, err);
    server.close();
    if (err.code === 'ENOTFOUND') {
      debug('ignore dns ENOTFOUND error, get free %s:%s', hostname, port);
      return callback(null, port);
    }
    return callback(err);
  });

  server.listen(port, hostname, () => {
    port = server.address().port;
    server.close();
    debug('get free %s:%s', hostname, port);
    return callback(null, port);
  });
}

listen 是使用 nodeJs 中 net 模块，来判断端口号是否可用

const server = new net.Server();

server.listen 如果没问题，就会 return callback(null, port)；

如果出错，就会 return callback(err); 再次进入 handleError 中，再做处理。
